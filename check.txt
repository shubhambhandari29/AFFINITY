-- 1) Create or reset sequence to MAX(ID)+1
DECLARE @next_id INT = ISNULL((SELECT MAX(ID) FROM dbo.tblUsers), 0) + 1;

IF OBJECT_ID('dbo.Seq_tblUsers_ID', 'SO') IS NULL
BEGIN
    EXEC('CREATE SEQUENCE dbo.Seq_tblUsers_ID AS INT START WITH ' + CAST(@next_id AS VARCHAR(20)) + ' INCREMENT BY 1');
END
ELSE
BEGIN
    EXEC('ALTER SEQUENCE dbo.Seq_tblUsers_ID RESTART WITH ' + CAST(@next_id AS VARCHAR(20)));
END

-- 2) Add default for ID (only once)
IF NOT EXISTS (
    SELECT 1
    FROM sys.default_constraints dc
    JOIN sys.columns c ON c.default_object_id = dc.object_id
    JOIN sys.tables t ON t.object_id = c.object_id
    WHERE t.name = 'tblUsers' AND c.name = 'ID'
)
BEGIN
    ALTER TABLE dbo.tblUsers
    ADD CONSTRAINT DF_tblUsers_ID DEFAULT (NEXT VALUE FOR dbo.Seq_tblUsers_ID) FOR ID;
END
