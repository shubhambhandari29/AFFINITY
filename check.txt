from __future__ import annotations

import argparse
from urllib.parse import quote

import requests
from azure.identity import ManagedIdentityCredential

GRAPH_SCOPE = "https://graph.microsoft.com/.default"
GRAPH_BASE_URL = "https://graph.microsoft.com/v1.0"


def get_graph_access_token() -> str:
    credential = ManagedIdentityCredential()
    return credential.get_token(GRAPH_SCOPE).token


def get_user_groups(user_id: str, access_token: str) -> list[dict]:
    groups: list[dict] = []
    url = (
        f"{GRAPH_BASE_URL}/users/{quote(user_id, safe='')}"
        "/transitiveMemberOf/microsoft.graph.group?$select=id,displayName"
    )
    headers = {"Authorization": f"Bearer {access_token}", "Accept": "application/json"}

    while url:
        response = requests.get(url, headers=headers, timeout=30)
        response.raise_for_status()
        payload = response.json()
        groups.extend(payload.get("value", []))
        url = payload.get("@odata.nextLink")

    return groups


def main() -> None:
    parser = argparse.ArgumentParser(description="Fetch all AD groups for a user.")
    parser.add_argument("--user-id", required=True, help="UPN or Entra object ID")
    args = parser.parse_args()

    token = get_graph_access_token()
    groups = get_user_groups(args.user_id, token)

    for g in groups:
        print(f"{g.get('displayName')}")


if __name__ == "__main__":
    main()
