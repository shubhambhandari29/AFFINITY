UPDATE A
SET TotalPrem = CONVERT(TEXT,
                        CONVERT(VARCHAR(50),
                                CAST(FLOOR(COALESCE(P.Premium,0) * 100) / 100.0 AS DECIMAL(18,2))))
FROM dbo.tblAcctSpecial A
OUTER APPLY
(
    SELECT SUM(PremiumAmt) AS Premium
    FROM dbo.tblPolicies P
    WHERE P.CustomerNum = A.CustomerNum
      AND P.PolicyStatus = 'Active'
) P
WHERE CAST(A.TotalPrem AS VARCHAR(50)) = '0.00';
