import { createSlice, createAsyncThunk } from "@reduxjs/toolkit";
import api from "../api/api";
export const ENV = process.env.REACT_APP_ENV;

export const loginThunk = createAsyncThunk(
  "auth/login",
  async (credentials, { rejectWithValue }) => {
    try {
      const { data } = await api.post(
        `/auth/${ENV === "local" ? "login" : "F5_login"}`,
        credentials,
      );
      return data;
    } catch (err) {
      return rejectWithValue(err.response?.data?.detail || "Login failed");
    }
  },
);

export const fetchCurrentUser = createAsyncThunk(
  "auth/me",
  async (_, { rejectWithValue }) => {
    try {
      // call the backend to validate cookie & return user info
      const { data } = await api.get("/auth/me", { withCredentials: true });
      return data;
    } catch (err) {
      return rejectWithValue(
        err.response?.data?.detail || "Failed to fetch user",
      );
    }
  },
);

export const refreshTokenThunk = createAsyncThunk(
  "auth/refresh",
  async (_, { rejectWithValue }) => {
    try {
      const { data } = await api.post("/auth/refresh");
      return data.token;
    } catch (err) {
      return rejectWithValue("Refresh failed");
    }
  },
);

export const logoutThunk = createAsyncThunk("auth/logout", async () => {
  try {
    await api.post("/auth/logout");
  } catch {}
});

const authSlice = createSlice({
  name: "auth",
  initialState: {
    user: null,
    accessToken: null,
    status: "idle",
    error: null,
  },
  reducers: {
    setAccessToken: (state, action) => {
      state.accessToken = action.payload;
    },
    logout: (state) => {
      state.user = null;
      state.accessToken = null;
      state.status = "idle";
    },
    updateUser: (state, action) => {
      state.user = action.payload;
    },
  },
  extraReducers: (builder) => {
    builder
      .addCase(loginThunk.pending, (state) => {
        state.status = "loggingin";
      })
      .addCase(loginThunk.fulfilled, (state, action) => {
        state.user = {
          ...action.payload.user,
          currentRole: action.payload.user.role.split(",")[0],
        };
        state.accessToken = action.payload.token;
        state.error = null;
        state.status = "idle";
      })
      .addCase(loginThunk.rejected, (state, action) => {
        state.error = action.payload;
        state.status = "idle";
      })
      .addCase(fetchCurrentUser.pending, (state) => {
        state.status = "loading";
      })
      .addCase(fetchCurrentUser.fulfilled, (state, action) => {
        state.user = {
          ...action.payload.user,
          currentRole: action.payload.user.role.split(",")[0],
        };
        state.accessToken = action.payload.token;
        state.error = null;
        state.status = "idle";
      })
      .addCase(fetchCurrentUser.rejected, (state, action) => {
        state.status = "idle";
      })
      .addCase(refreshTokenThunk.pending, (state) => {
        state.status = "loading";
      })
      .addCase(refreshTokenThunk.fulfilled, (state, action) => {
        state.accessToken = action.payload;
        state.status = "idle";
      })
      .addCase(refreshTokenThunk.rejected, (state) => {
        state.status = "idle";
        state.accessToken = null;
        state.user = null;
      })
      .addCase(logoutThunk.pending, (state) => {
        state.status = "loading";
      })
      .addCase(logoutThunk.fulfilled, (state) => {
        state.user = null;
        state.accessToken = null;
        state.status = "idle";
      });
  },
});

export const { setAccessToken, logout, updateUser } = authSlice.actions;
export default authSlice.reducer;

const express = require("express");
const path = require("path");
const app = express();

/**
 * 1. THE ECHO API ENDPOINT
 * This is the "Intercom" React will call to ask for the F5 headers.
 */
app.get("/api/user_info", (req, res) => {
  // Return the data as a clean JSON object
  res.json({
    user: req.header("username") || "",
    groups: req.header("groups") || "",
  });
});

/**
 * 2. SERVE STATIC FILES
 */
const buildPath = path.join(__dirname, "build");
app.use(express.static(buildPath));

/**
 * 3. SPA ROUTING SUPPORT
 * If a user refreshes the page on a route like /profile,
 * this ensures they get index.html instead of a 404.
 */
app.get(/.*/, (req, res) => {
  res.sendFile(path.join(buildPath, "index.html"));
});

/**
 * 4. START THE SERVER
 * Azure App Service automatically assigns a PORT via environment variables.
 */
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
